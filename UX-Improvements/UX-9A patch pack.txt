from __future__ import annotations

from dataclasses import dataclass
from typing import List, Dict, Any, Tuple
import math


@dataclass
class ClusterSummary:
    cluster_id: str
    label: str
    count: int
    avg_embedding: list[float] | None = None
    first_seen_ts: str | None = None
    last_seen_ts: str | None = None
    unnamed: bool = False


class PeopleMergeService:
    """
    UX-9A:
    Lightweight merge-intelligence backend.
    Produces ranked merge suggestions and unnamed-cluster payloads.
    """

    def __init__(self, review_store=None):
        self.review_store = review_store

    def build_people_quick_payload(self, clusters: List[Dict[str, Any]]) -> Dict[str, Any]:
        normalized = self._normalize_clusters(clusters)

        top_people = sorted(normalized, key=lambda x: x.count, reverse=True)[:8]
        unnamed_count = sum(1 for c in normalized if c.unnamed)

        suggestions = self.build_merge_suggestions(clusters)
        merge_candidates = len(suggestions)

        return {
            "top_people": [
                {
                    "id": c.cluster_id,
                    "label": c.label,
                    "count": c.count,
                }
                for c in top_people
            ],
            "merge_candidates": merge_candidates,
            "unnamed_count": unnamed_count,
        }

    def build_merge_suggestions(self, clusters: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        normalized = self._normalize_clusters(clusters)
        suggestions: List[Dict[str, Any]] = []

        for i in range(len(normalized)):
            for j in range(i + 1, len(normalized)):
                left = normalized[i]
                right = normalized[j]

                if left.cluster_id == right.cluster_id:
                    continue

                if self._is_rejected(left.cluster_id, right.cluster_id):
                    continue

                score = self._score_pair(left, right)
                if score < 0.58:
                    continue

                if not self._passes_guardrails(left, right, score):
                    continue

                suggestions.append({
                    "left_id": left.cluster_id,
                    "right_id": right.cluster_id,
                    "left_label": left.label,
                    "right_label": right.label,
                    "score": score,
                    "label": f"{left.label} ↔ {right.label} (score={score:.2f})",
                    "left_count": left.count,
                    "right_count": right.count,
                    "left_unnamed": left.unnamed,
                    "right_unnamed": right.unnamed,
                })

        suggestions.sort(key=lambda x: x["score"], reverse=True)
        return suggestions[:20]

    def build_unnamed_clusters(self, clusters: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        normalized = self._normalize_clusters(clusters)
        unnamed = [c for c in normalized if c.unnamed]
        unnamed.sort(key=lambda x: x.count, reverse=True)

        return [
            {
                "id": c.cluster_id,
                "label": c.label,
                "count": c.count,
            }
            for c in unnamed
        ]

    def accept_merge(self, left_id: str, right_id: str):
        if self.review_store:
            self.review_store.record_people_merge_decision(left_id, right_id, "accepted")

    def reject_merge(self, left_id: str, right_id: str):
        if self.review_store:
            self.review_store.record_people_merge_decision(left_id, right_id, "rejected")

    def _normalize_clusters(self, clusters: List[Dict[str, Any]]) -> List[ClusterSummary]:
        normalized: List[ClusterSummary] = []

        for item in list(clusters or []):
            if not isinstance(item, dict):
                continue

            cluster_id = item.get("id") or item.get("person_id") or item.get("label")
            if cluster_id is None:
                continue

            label = item.get("label") or item.get("name") or str(cluster_id)
            count = int(item.get("count", 0) or 0)
            avg_embedding = item.get("avg_embedding")
            first_seen_ts = item.get("first_seen_ts")
            last_seen_ts = item.get("last_seen_ts")

            unnamed = False
            lower_label = str(label).lower()
            if lower_label.startswith("face_") or "unnamed" in lower_label or "unknown" in lower_label:
                unnamed = True

            normalized.append(
                ClusterSummary(
                    cluster_id=str(cluster_id),
                    label=str(label),
                    count=count,
                    avg_embedding=avg_embedding,
                    first_seen_ts=first_seen_ts,
                    last_seen_ts=last_seen_ts,
                    unnamed=unnamed,
                )
            )

        return normalized

    def _score_pair(self, left: ClusterSummary, right: ClusterSummary) -> float:
        score = 0.0

        emb_sim = self._embedding_similarity(left.avg_embedding, right.avg_embedding)
        if emb_sim is not None:
            score += 0.70 * emb_sim

        size_balance = self._size_balance(left.count, right.count)
        score += 0.10 * size_balance

        if left.unnamed or right.unnamed:
            score += 0.08

        temporal = self._temporal_overlap_score(left.first_seen_ts, left.last_seen_ts, right.first_seen_ts, right.last_seen_ts)
        score += 0.12 * temporal

        return max(0.0, min(1.0, score))

    def _passes_guardrails(self, left: ClusterSummary, right: ClusterSummary, score: float) -> bool:
        # avoid suggesting merges between two large, likely distinct identities unless score is strong
        if left.count >= 12 and right.count >= 12 and score < 0.75:
            return False

        # avoid merging two tiny clusters if no embedding signal exists
        if left.avg_embedding is None and right.avg_embedding is None and left.count <= 2 and right.count <= 2:
            return False

        return True

    def _embedding_similarity(self, a, b) -> float | None:
        if not a or not b:
            return None
        try:
            if len(a) != len(b):
                return None
            dot = sum(float(x) * float(y) for x, y in zip(a, b))
            na = math.sqrt(sum(float(x) * float(x) for x in a))
            nb = math.sqrt(sum(float(y) * float(y) for y in b))
            if na == 0 or nb == 0:
                return None
            cos = dot / (na * nb)
            # map cosine-ish similarity into 0..1
            return max(0.0, min(1.0, (cos + 1.0) / 2.0))
        except Exception:
            return None

    def _size_balance(self, a: int, b: int) -> float:
        if a <= 0 or b <= 0:
            return 0.0
        bigger = max(a, b)
        smaller = min(a, b)
        return smaller / bigger

    def _temporal_overlap_score(self, a_start, a_end, b_start, b_end) -> float:
        # lightweight placeholder until better temporal reasoning exists
        if not a_start or not a_end or not b_start or not b_end:
            return 0.3
        if a_start <= b_end and b_start <= a_end:
            return 1.0
        return 0.2

    def _is_rejected(self, left_id: str, right_id: str) -> bool:
        if not self.review_store:
            return False
        return self.review_store.has_people_merge_rejection(left_id, right_id)

2.2 Add helper methods

Add these methods to the DB class:

def _normalize_merge_pair(self, left_id: str, right_id: str):
    a = str(left_id)
    b = str(right_id)
    return (a, b) if a <= b else (b, a)

def record_people_merge_decision(self, left_id: str, right_id: str, decision: str):
    left_id, right_id = self._normalize_merge_pair(left_id, right_id)

    conn = self._get_connection() if hasattr(self, "_get_connection") else self.conn
    cur = conn.cursor()
    cur.execute(
        """
        INSERT INTO people_merge_review (left_cluster_id, right_cluster_id, decision)
        VALUES (?, ?, ?)
        """,
        (left_id, right_id, decision),
    )
    conn.commit()

def has_people_merge_rejection(self, left_id: str, right_id: str) -> bool:
    left_id, right_id = self._normalize_merge_pair(left_id, right_id)

    conn = self._get_connection() if hasattr(self, "_get_connection") else self.conn
    cur = conn.cursor()
    cur.execute(
        """
        SELECT 1
        FROM people_merge_review
        WHERE left_cluster_id = ? AND right_cluster_id = ? AND decision = 'rejected'
        ORDER BY id DESC
        LIMIT 1
        """,
        (left_id, right_id),
    )
    return cur.fetchone() is not None

    Adapt only the source extraction here if your internal structure changes.

            "avg_embedding": item.get("avg_embedding"),

    return out

from PySide6.QtCore import Signal
from PySide6.QtWidgets import (
    QWidget, QVBoxLayout, QLabel, QListWidget, QListWidgetItem,
    QPushButton, QHBoxLayout, QDialog, QDialogButtonBox, QGroupBox
)


class PeopleMergeSuggestionsPanel(QWidget):
    mergeAccepted = Signal(str, str)
    mergeRejected = Signal(str, str)

    def __init__(self, parent=None):
        super().__init__(parent)

        self.lbl_title = QLabel("Possible People Merges")

        self.list_widget = QListWidget()

        self.left_preview = QLabel("Left cluster preview")
        self.right_preview = QLabel("Right cluster preview")

        left_box = QGroupBox("Left")
        left_layout = QVBoxLayout(left_box)
        left_layout.addWidget(self.left_preview)

        right_box = QGroupBox("Right")
        right_layout = QVBoxLayout(right_box)
        right_layout.addWidget(self.right_preview)

        preview_row = QHBoxLayout()
        preview_row.addWidget(left_box)
        preview_row.addWidget(right_box)

        self.btn_accept = QPushButton("Merge Selected")
        self.btn_reject = QPushButton("Reject Selected")

        btn_row = QHBoxLayout()
        btn_row.addWidget(self.btn_accept)
        btn_row.addWidget(self.btn_reject)

        layout = QVBoxLayout(self)
        layout.addWidget(self.lbl_title)
        layout.addWidget(self.list_widget)
        layout.addLayout(preview_row)
        layout.addLayout(btn_row)

        self._items = []

        self.btn_accept.clicked.connect(self._accept_selected)
        self.btn_reject.clicked.connect(self._reject_selected)
        self.list_widget.currentItemChanged.connect(self._on_current_changed)

    def set_suggestions(self, suggestions):
        self._items = list(suggestions or [])
        self.list_widget.clear()

        for item in self._items:
            left_id = str(item.get("left_id", ""))
            right_id = str(item.get("right_id", ""))
            score = item.get("score")
            score_txt = f"{score:.2f}" if isinstance(score, (float, int)) else "?"
            title = item.get("label") or f"{left_id} ↔ {right_id}  (score={score_txt})"

            list_item = QListWidgetItem(title)
            list_item.setData(256, item)
            self.list_widget.addItem(list_item)

        if self.list_widget.count() > 0:
            self.list_widget.setCurrentRow(0)

    def _on_current_changed(self, current, previous):
        if not current:
            self.left_preview.setText("Left cluster preview")
            self.right_preview.setText("Right cluster preview")
            return

        item = current.data(256) or {}
        self.left_preview.setText(
            f"ID: {item.get('left_id', '')}\n"
            f"Label: {item.get('left_label', '')}\n"
            f"Count: {item.get('left_count', 0)}"
        )
        self.right_preview.setText(
            f"ID: {item.get('right_id', '')}\n"
            f"Label: {item.get('right_label', '')}\n"
            f"Count: {item.get('right_count', 0)}"
        )

    def _accept_selected(self):
        item = self.list_widget.currentItem()
        if not item:
            return
        payload = item.data(256) or {}
        self.mergeAccepted.emit(str(payload.get("left_id", "")), str(payload.get("right_id", "")))

    def _reject_selected(self):
        item = self.list_widget.currentItem()
        if not item:
            return
        payload = item.data(256) or {}
        self.mergeRejected.emit(str(payload.get("left_id", "")), str(payload.get("right_id", "")))
