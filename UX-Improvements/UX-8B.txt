Here is the **exact UX-8B patch pack**.

## UX-8B goal

Polish the **result pane interaction** so it feels connected to the shell and less transitional.

This phase focuses on:

* preserving previous results while search is running,
* showing better loading / no-results / no-project states,
* surfacing “why these results”,
* making chip / filter / result transitions feel stable,
* improving result-pane state handling without rewriting the whole grid.

This is the correct next step after UX-8A.

---

# Scope

This pack adds:

1. richer result-pane state support in `google_layout.py`
2. lightweight “why these results” support in the header
3. previous-results preservation in the search controller/store
4. better empty-state transitions
5. safer search-state-driven UI refresh
6. optional result-explanation label

It does **not** yet do:

* animation framework
* skeleton thumbnail components
* full result virtualization rewrite
* semantic ranking explanation cards per image

## 2.2 Patch `apply_result_summary()`

Replace the full method with:

```python
    def apply_result_summary(
        self,
        result_paths=None,
        result_count: int = 0,
        result_facets=None,
        family: Optional[str] = None,
        warnings=None,
    ):
        state = self.store.get_state()
        warning_list = list(warnings or [])
        model_warning = ""

        joined = " ".join(str(w) for w in warning_list).lower()
        if "clip-vit-large-patch14" in joined or "better model available" in joined:
            model_warning = "Better model available"

        discover_counts = dict(state.discover_counts or {})
        discover_previews = dict(state.discover_previews or {})
        new_paths = list(result_paths or [])

        if state.preset_id:
            discover_counts[state.preset_id] = result_count

            preview_labels = []
            for path in new_paths[:3]:
                try:
                    import os
                    preview_labels.append(os.path.basename(path))
                except Exception:
                    pass
            discover_previews[state.preset_id] = preview_labels

        explanation_parts = []
        if state.preset_id:
            explanation_parts.append(state.preset_id.replace("_", " ").title())
        if state.query_text.strip():
            explanation_parts.append(f'query "{state.query_text.strip()}"')
        if state.active_people:
            explanation_parts.append(f"{len(state.active_people)} people filter(s)")
        if state.active_filters:
            explanation_parts.append(f"{len(state.active_filters)} active filter(s)")

        result_explanation = ""
        if explanation_parts:
            result_explanation = "Results based on " + ", ".join(explanation_parts)

        self.store.update(
            result_paths=new_paths,
            displayed_result_paths=new_paths,
            last_nonempty_result_paths=new_paths if new_paths else list(state.last_nonempty_result_paths or []),
            result_count=result_count,
            result_facets=result_facets or {},
            family=family,
            warnings=warning_list,
            model_warning=model_warning,
            discover_counts=discover_counts,
            discover_previews=discover_previews,
            search_in_progress=False,
            empty_state_reason=None if result_count > 0 else "no_results",
            intent_summary=self._build_intent_summary(),
            result_explanation=result_explanation,
        )
```

## Why

This adds stable display paths, preserves last nonempty results, and creates a human-readable explanation string.

        top_row.addWidget(self.cmb_sort)

        outer = QVBoxLayout(self)
        outer.setContentsMargins(0, 0, 0, 0)
        outer.setSpacing(4)
        outer.addLayout(top_row)
        outer.addWidget(self.lbl_explanation)

        self.store.stateChanged.connect(self._on_state_changed)
        self.cmb_sort.currentIndexChanged.connect(self._on_sort_changed)

    def _on_state_changed(self, state):
        if not isValid(self):
            return

        if state.onboarding_mode:
            self.lbl_summary.setText("No active project")
            self.lbl_count.setText("")
            self.lbl_status.setVisible(False)
            self.lbl_warning.setVisible(False)
            self.lbl_model.setVisible(False)
            self.lbl_explanation.setVisible(False)
            return

        if state.search_in_progress:
            self.lbl_summary.setText(state.intent_summary or "Searching...")
            self.lbl_status.setText("Searching")
            self.lbl_status.setVisible(True)
        else:
            self.lbl_summary.setText(state.intent_summary or "All Photos")
            self.lbl_status.setVisible(False)

        self.lbl_count.setText(f"{state.result_count} result(s)")

        warnings = list(getattr(state, "warnings", []) or [])
        if warnings:
            self.lbl_warning.setText(str(warnings[0]))
            self.lbl_warning.setVisible(True)
        else:
            self.lbl_warning.setVisible(False)

        model_warning = getattr(state, "model_warning", "")
        if model_warning:
            self.lbl_model.setText(model_warning)
            self.lbl_model.setVisible(True)
        else:
            self.lbl_model.setVisible(False)

        explanation = getattr(state, "result_explanation", "")
        if explanation:
            self.lbl_explanation.setText(explanation)
            self.lbl_explanation.setVisible(True)
        else:
            self.lbl_explanation.setVisible(False)

    def _on_sort_changed(self):
        if not self.controller:
            return

        state = self.store.get_state()
        state.sort_mode = self.cmb_sort.currentData()
        self.store.stateChanged.emit(state)
        self.controller.run_search()

```python
from PySide6.QtCore import Signal
from PySide6.QtWidgets import QWidget, QVBoxLayout, QLabel, QPushButton


class EmptyStateView(QWidget):
    actionRequested = Signal(str)

    def __init__(self, parent=None):
        super().__init__(parent)

        self.lbl_title = QLabel("No results")
        self.lbl_title.setObjectName("EmptyStateTitle")

        self.lbl_message = QLabel("")
        self.lbl_message.setWordWrap(True)
        self.lbl_message.setObjectName("EmptyStateMessage")

        self.btn_primary = QPushButton()
        self.btn_primary.setVisible(False)
        self.btn_primary.clicked.connect(self._emit_primary)

        self._primary_action = ""

        layout = QVBoxLayout(self)
        layout.addStretch(1)
        layout.addWidget(self.lbl_title)
        layout.addWidget(self.lbl_message)
        layout.addWidget(self.btn_primary)
        layout.addStretch(1)

        self.set_state("no_results")

    def _emit_primary(self):
        if self._primary_action:
            self.actionRequested.emit(self._primary_action)

    def set_state(self, reason: str, warnings=None):
        warnings = list(warnings or [])

        if reason == "no_project":
            self.lbl_title.setText("No active project")
            self.lbl_message.setText("Create or select a project to begin browsing and searching.")
            self._primary_action = "create_or_select_project"
            self.btn_primary.setText("Select Project")
            self.btn_primary.setVisible(True)

        elif reason == "embeddings_missing":
            self.lbl_title.setText("Embeddings not ready")
            self.lbl_message.setText("Smart search quality will improve after extracting embeddings.")
            self._primary_action = "extract_embeddings"
            self.btn_primary.setText("Extract Embeddings")
            self.btn_primary.setVisible(True)

        elif reason == "indexing":
            self.lbl_title.setText("Indexing in progress")
            self.lbl_message.setText("Background indexing is still running. Results may improve as processing completes.")
            self._primary_action = ""
            self.btn_primary.setVisible(False)

        elif reason == "loading":
            self.lbl_title.setText("Searching")
            self.lbl_message.setText("Preparing refined results. Previous results may remain visible until the new search completes.")
            self._primary_action = ""
            self.btn_primary.setVisible(False)

        else:
            self.lbl_title.setText("No results")
            extra = f"\n\n{warnings[0]}" if warnings else ""
            self.lbl_message.setText("No matches were found. Try a broader query, remove some filters, or switch presets." + extra)
            self._primary_action = ""
            self.btn_primary.setVisible(False)
```

## 5.2 Replace `_on_search_state_changed()` with this version:

```python
def _on_search_state_changed(self, state):
    try:
        if not hasattr(self, "empty_state"):
            return

        displayed_paths = list(getattr(state, "displayed_result_paths", []) or [])
        result_paths = list(getattr(state, "result_paths", []) or [])
        search_in_progress = bool(getattr(state, "search_in_progress", False))

        # Onboarding / no project
        if getattr(state, "onboarding_mode", False):
            self._show_empty_state("no_project", getattr(state, "warnings", []))
            return

        # Search in progress: keep showing previous results if available
        if search_in_progress:
            if displayed_paths:
                self._show_results_surface()
            else:
                self._show_empty_state("loading", getattr(state, "warnings", []))
            return

        # Indexing state
        if getattr(state, "indexing_in_progress", False) and not result_paths:
            self._show_empty_state("indexing", getattr(state, "warnings", []))
            return

        # Missing embeddings path
        if not getattr(state, "embeddings_ready", True):
            reason = getattr(state, "empty_state_reason", None)
            if reason == "embeddings_missing" and not result_paths:
                self._show_empty_state("embeddings_missing", getattr(state, "warnings", []))
                return

        if not result_paths:
            self._show_empty_state(getattr(state, "empty_state_reason", None) or "no_results",
                                   getattr(state, "warnings", []))
            return

        self._show_results_surface()

    except Exception as e:
        print(f"[GooglePhotosLayout] _on_search_state_changed error: {e}")
```

## Why

This is the core UX-8B behavior: preserve previous results during search instead of flashing empty state.
